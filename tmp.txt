
	private Logger logger = LoggerFactory.getLogger(OrderController.class);
	@Autowired
	private ProductServiceClient productServiceClient;
	
	@RequestMapping(method = RequestMethod.POST)
	public OrderResponseDTO createOrder(@RequestBody OrderRequestDTO orderRequest) {

		logger.info("New Order  Create Request {} ", orderRequest);

		ProductDTO productDetail = productServiceClient.readProductById(orderRequest.getProductNo());

		OrderResponseDTO responseDTO = new OrderResponseDTO();

		responseDTO.setMessage(
				"Your order is created product: " + productDetail.getName() + ", price" + productDetail.getPrice());
		responseDTO.setOrderNo(UUID.randomUUID().toString());
		responseDTO.setStatus("OK");
		logger.info("Order Result {} ", responseDTO);
		return responseDTO;

	}
